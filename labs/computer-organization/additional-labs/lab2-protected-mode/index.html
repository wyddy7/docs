/* –ü—Ä–æ—Ç–µ—Å—Ç–∏—Ä—É–µ–º –∑–∞–ø–∏—Å—å —Å–∏–º–≤–æ–ª–æ–≤ ascii –≤ –≤–∏–¥–µ–æ–ø–∞–º—è—Ç—å */

  movl $320, %ebx			// –Ω–∞—á–∞–ª—å–Ω–æ–µ —Å–º–µ—â–µ–Ω–∏–µ –Ω–∞ —ç–∫—Ä–∞–Ω–µ
  movl $110, %ecx			// —á–∏—Å–ª–æ –≤—ã–≤–æ–¥–∏–º—ã—Ö —Å–∏–º–≤–æ–ª–æ–≤
  movw %ds:DS_OFF(symbol), %ax		// –∑–∞–≥—Ä—É–∂–∞–µ–º –Ω–∞—á–∞–ª—å–Ω—ã–π —Å–∏–º–≤–æ–ª —Å –∞—Ç—Ä–∏–±—É—Ç–æ–º (–æ–±—Ä–∞—â–µ–Ω–∏–µ –∫ —Å–µ–≥–º–µ–Ω—Ç—É –¥–∞–Ω–Ω—ã—Ö!)
print_pm:
  movw %ax, %es:(%ebx)			// –∑–∞–ø–∏—Å—ã–≤–∞–µ–º –æ—á–µ—Ä–µ–¥–Ω–æ–π —Å–∏–º–≤–æ–ª –≤ –≤–∏–¥–µ–æ–ø–∞–º—è—Ç—å (–æ–±—Ä–∞—â–µ–Ω–∏–µ –∫ —Å–µ–≥–º–µ–Ω—Ç—É –≤–∏–¥–µ–æ–ø–∞–º—è—Ç–∏!)
  addl $2, %ebx				// —Å–º–µ—â–∞–µ–º—Å—è –≤ –≤–∏–¥–µ–æ–±—É—Ñ–µ—Ä–µ
  incw %ax				// –∫–æ–¥ —Å–ª–µ–¥—É—é—â–µ–≥–æ —Å–∏–º–≤–æ–ª–∞
  pushl %ecx				// —Å–æ—Ö—Ä–∞–Ω—è–µ–º ecx –≤ —Å—Ç–µ–∫–µ (–æ–±—Ä–∞—â–µ–Ω–∏–µ –∫ —Å–µ–≥–º–µ–Ω—Ç—É —Å—Ç–µ–∫–∞!)
  movl $0xFFFFFF, %ecx			// –¥–æ–±–∞–≤–∏–º –ø–∞—É–∑—É –¥–ª—è –Ω–∞–≥–ª—è–¥–Ω–æ—Å—Ç–∏

  nop
  loop idle				// —Ü–∏–∫–ª –ø–∞—É–∑—ã
  popl %ecx				// –≤–æ–∑–≤—Ä–∞—â–∞–µ–º ecx –∏–∑ —Å—Ç–µ–∫–∞
  loop print_pm				// —Ü–∏–∫–ª –≤—ã–≤–æ–¥–∞ –Ω–∞ —ç–∫—Ä–∞–Ω

/* –ü–æ–¥–≥–æ—Ç–æ–≤–∏–º –ø–µ—Ä–µ—Ö–æ–¥ –≤ —Ä–µ–∞–ª—å–Ω—ã–π —Ä–µ–∂–∏–º. –°–ø–µ—Ä–≤–∞ –Ω–∞–¥–æ —Å—Ñ–æ—Ä–º–∏—Ä–æ–≤–∞—Ç—å –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä—ã –¥–ª—è —Ä–µ–∞–ª—å–Ω–æ–≥–æ —Ä–µ–∂–∏–º–∞.

  movb $0, %ds:DS_OFF(seg_text) + 6 // –æ–±–Ω—É–ª—è–µ–º —Å—Ç–∞—Ä—à—É—é —á–∞—Å—Ç—å –ø—Ä–µ–¥–µ–ª–∞
  orb $EXEC_READ, %ds:DS_OFF(seg_text) + 5	// –∏ –¥–æ—Å—Ç—É–ø–Ω—ã–π –¥–ª—è —á—Ç–µ–Ω–∏—è

						// –î–ª—è –æ—Å—Ç–∞–ª—å–Ω—ã—Ö —Å–µ–≥–º–µ–Ω—Ç–æ–≤ —Ç—Ä–µ–±–æ–≤–∞–Ω–∏—è —Å–ª–µ–¥—É—é—â–∏–µ

						// 	Writable (W = 1)
  movw $0xFFFF, %ds:DS_OFF(seg_stack)		// 	Present (P = 1)
  movb $0, %ds:DS_OFF(seg_stack) + 6
  orb $READ_WRITE, %ds:DS_OFF(seg_stack) + 5

  movw $0xFFFF, %ds:DS_OFF(seg_videomem)
  movb $0, %ds:DS_OFF(seg_videomem) + 6
  orb $READ_WRITE, %ds:DS_OFF(seg_videomem) + 5

  movw $0x10, %ax			// –∑–∞–≥—Ä—É–∑–∏–º —Ç–µ–Ω–µ–≤—ã–µ —Ä–µ–≥–∏—Å—Ç—Ä—ã —Å–µ–≥–º–µ–Ω—Ç–æ–≤, —Ç. –∫. –∏–∑–º–µ–Ω–∏–ª–∏—Å—å —Å–∞–º–∏ –¥–µ—Å–∫—Ä–∏–ø—Ç–æ—Ä—ã
  movw %ax, %ds
  movw $0x18, %ax
  movw %ax, %ss
  movw $0x20, %ax
  movw %ax, %es

  ljmp $0x8, $CS_OFF(goto_64k_segment)	// –≤—ã–ø–æ–ª–Ω–∏–º –¥–∞–ª—å–Ω–∏–π –ø–µ—Ä–µ—Ö–æ–¥, —á—Ç–æ–±—ã —Å–Ω–æ–≤–∞ –∑–∞–≥—Ä—É–∑–∏—Ç—å —Å–µ–ª–µ–∫—Ç–æ—Ä
					// –≤ CS –∏ –º–æ–¥–∏—Ñ–∏—Ü–∏—Ä–æ–≤–∞—Ç—å –µ–≥–æ —Ç–µ–Ω–µ–≤–æ–π —Ä–µ–≥–∏—Å—Ç—Ä

goto_64k_segment:

  .code16				// —Å–Ω–æ–≤–∞ –≥–µ–Ω–µ—Ä–∏—Ä—É–µ–º 16-—Ä–∞–∑—Ä—è–¥–Ω—ã–π –∫–æ–¥

  andl $0xFFFFFFFE, %eax
  movl %eax, %cr0			// —Å–±—Ä–æ—Å–∏–º –±–∏—Ç PE –≤ —Ä–µ–≥–∏—Å—Ç—Ä–µ CR0
