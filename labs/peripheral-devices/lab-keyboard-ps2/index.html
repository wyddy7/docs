	mov r9, zero

	movia r10, HEX_RESET_DELAY

	# –°–±—Ä–æ—Å —Å–æ—Å—Ç–æ—è–Ω–∏—è, –µ—Å–ª–∏ –∑–∞–¥–µ—Ä–∂–∫–∞ —Ä–∞–≤–Ω–∞ 0.
	ble r10, zero, reset_hex

	call READ_PS2_DATA

	mov r5, r9

	mov r8, r2
	mov r9, r3
	# –û—á–∏—Å—Ç–∫–∞ 7-—Å–µ–≥–º–µ–Ω—Ç–Ω—ã—Ö –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–æ–≤.

	ldwio r2, (r2)
	ret

/* –î–æ–±–∞–≤–ª—è–µ—Ç –±–∞–π—Ç –≤ –∫–æ–Ω–µ—Ü (–ª–µ–≤—É—é —á–∞—Å—Ç—å) –∏—Å—Ö–æ–¥–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è. */
APPEND_BYTE:
	# –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	stw r8, (sp)
	mov r2, r4
	mov r3, r5
	movi r8, 3
	# –ï—Å–ª–∏ –¥–ª–∏–Ω–∞ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ < 4, –ø—Ä–æ–ø—É—Å–∫–∞–µ–º —Å–º–µ—â–µ–Ω–∏–µ.

	# –°–º–µ—â–∞–µ–º –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –≤–ª–µ–≤–æ.
	srli r2, r2, 8
	# –î–µ–∫—Ä–µ–º–µ–Ω—Ç–∏—Ä—É–µ–º –¥–ª–∏–Ω—É –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏.

	# –û–ø—Ä–µ–¥–µ–ª—è–µ–º —Å–º–µ—â–µ–Ω–∏–µ.
	muli r8, r3, 8
	# –°–º–µ—â–∞–µ–º –¥–æ–±–∞–≤–ª—è–µ–º—ã–π –±–∞–π—Ç –≤–ª–µ–≤–æ.
	sll r8, r6, r8

	or r2, r2, r8
	# –ò–Ω–∫—Ä–µ–º–µ–Ω—Ç–∏—Ä—É–µ–º –¥–ª–∏–Ω—É –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏.
	addi r3, r3, 1
	# –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	ldw r8, (sp)
	ret

/* –û—á–∏—â–∞–µ—Ç 7-—Å–µ–≥–º–µ–Ω—Ç–Ω—ã–µ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã. */
CLEAR_HEX_DISPLAY:
	# –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	stw r8, (sp)
	# –û—á–∏—Å—Ç–∫–∞ –ø–µ—Ä–≤–æ–π —á–µ—Ç–≤–µ—Ä–∫–∏ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–æ–≤.
	movia r8, HEX_DATA_L
	stwio zero, (r8)
	# –û—á–∏—Å—Ç–∫–∞ –ø–µ—Ä–≤–æ–π —á–µ—Ç–≤–µ—Ä–∫–∏ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–æ–≤.
	movia r8, HEX_DATA_H
	stwio zero, (r8)
	# –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	ldw r8, (sp)
	ret

/*	–í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –∞–¥—Ä–µ—Å 7-—Å–µ–≥–º–µ–Ω—Ç–Ω–æ–≥–æ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞. */

	# –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.

	stw r9, -4(sp)

	# –í—ã–±–æ—Ä —Ä–µ–≥–∏—Å—Ç—Ä–∞ –¥–∞–Ω–Ω—ã—Ö.
	movi r9, 3
	bgtu r8, r9, HEX_DISPLAY_ADDR_H 
HEX_DISPLAY_ADDR_L:
	movia r2, HEX_DATA_L
	add r2, r2, r4
	br HEX_DISPLAY_ADDR_END
HEX_DISPLAY_ADDR_H:
	movia r2, HEX_DATA_H
	add r2, r2, r4
	subi r2, r2, 4
HEX_DISPLAY_ADDR_END:
	# –í–æ—Å—Å—Ç–∞–Ω–æ–≤–ª–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	ldw r8, (sp)
	ldw r9, -4(sp)
	ret

/*	–í—ã–≤–æ–¥–∏—Ç —Ü–∏—Ñ—Ä—ã –Ω–∞ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã. */
HEX_DISPLAY_DIGIT:
    # –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	stw ra, (sp)
	stw r2, -4(sp)
	stw r8, -8(sp)
	stw r9, -12(sp)
	subi sp, sp, 16
	# –ü–æ–ª—É—á–µ–Ω–∏–µ –∞–¥—Ä–µ—Å–∞ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞ –≤ r2.
	mov r8, r4
	mov r4, r5
	call HEX_DISPLAY_ADDR
	mov r4, r8
	# –ú–∞—Å–∫–∞ –Ω–∞ 16-—é —Ü–∏—Ñ—Ä—É.
	andi r8, r4, 0xF
	# –ü–æ–ª—É—á–µ–Ω–∏–µ –∑–Ω–∞—á–µ–Ω–∏—è –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è —Ü–∏—Ñ—Ä—ã –Ω–∞ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–µ.
	movia r9, HEX_DIGITS
	add r8, r8, r9
	ldb r8, (r8)
	call HEX_SAVE
	# –û—Ç–æ–±—Ä–∞–∂–µ–Ω–∏–µ –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç–∏ —Ü–∏—Ñ—Ä –Ω–∞ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–µ.
	stwio r8, (r2)
	# –í–æ—Å—Å—Ç–∞–≤–Ω–æ–≤–ª–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	addi sp, sp, 16
	ldw ra, (sp)
	ldw r2, -4(sp)
	ldw r8, -8(sp)
	ldw r9, -12(sp)
	ret
	
/*	–í—ã–≤–æ–¥–∏—Ç –∑–Ω–∞—á–µ–Ω–∏–µ –Ω–∞ 7-—Å–µ–≥–º–µ–Ω—Ç–Ω—ã–µ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä—ã. */

	# –°–æ—Ö—Ä–∞–Ω–µ–Ω–∏–µ —Ä–µ–≥–∏—Å—Ç—Ä–æ–≤.
	stw ra, (sp)
	stw r4, -4(sp)
	stw r5, -8(sp)
	stw r8, -12(sp)
	subi sp, sp, 16
	# r8 —Ö—Ä–∞–Ω–∏—Ç –∫–æ–ª-–≤–æ –æ—Ç–æ–±—Ä–∞–∂–∞–µ–º—ã—Ö —Ü–∏—Ñ—Ä.
	mov r8, r5
	# r5 —Ö—Ä–∞–Ω–∏—Ç —Ç–µ–∫—É—â–∏–π –Ω–æ–º–µ—Ä –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞.
	mov r5, zero

	# –¶–∏–∫–ª –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è —Ü–∏—Ñ—Ä.
	beq r5, r8, HEX_DISPLAY_END
	call HEX_DISPLAY_DIGIT

	addi r5, r5, 1
	br HEX_DISPLAY_LOOP

	addi sp, sp, 16
	ldw ra, (sp)
	ldw r4, -4(sp)
	ldw r5, -8(sp)
	ldw r8, -12(sp)
	ret

/* –í–æ–∑–≤—Ä–∞—â–∞–µ—Ç –ø–æ—Å–ª–µ–¥–æ–≤–∞—Ç–µ–ª—å–Ω–æ—Å—Ç—å –∑–Ω–∞—á–µ–Ω–∏–π –¥–ª—è –æ—Ç–æ–±—Ä–∞–∂–µ–Ω–∏—è –Ω–∞ –∏–Ω–¥–∏–∫–∞—Ç–æ—Ä–∞—Ö */

	stw ra, (sp)
	stw r4, -4(sp)
	subi sp, sp, 8
